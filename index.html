Find the Smaller Angle

var Minimal_Angle = (h, m) => 
{
      h=h%12;
      let min=(m*360)/60;
      let hour=(h*360)/12+(m*360)/(12*60);
      let angle=Math.abs(hour-min);
      
      if(angle>180){
        return 360-angle;
      }
      else{
        return angle;
      }
};

var Check_Leap = (year) => 
{
   if(year%4===0 && year%100===0 && year%400===0 ){
    return ("Leap Year");
 }
    else if(year%4===0 && year%100!==0 && year%400!==0 ){
      return ("Leap Year");
    }
else{
   return ("Non Leap Year");
}
};

Perfect Number Check.

 
var Perfect_Check = (N) => 
{
  let sum=0;
  for(i=0;i< N;i++){
    if(N%i===0){
      sum+=i;
    }
  }
    if(sum===N){
      return "YES";
    }  
    else if(N===1){
      return "YES";
    }
    else{
      return "NO";
    }
        
};

var Reverse_Number = (N) => 
{
  let rev=0;
while(N>0){
    digit=N%10;
